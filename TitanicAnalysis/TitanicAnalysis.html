def make_pictures(key):   
    data.groupby(key)['PassengerId'].count().plot(kind='pie',autopct='%0.f%%')
    plt.title(key + ' VS Count')
    plt.show()
    
    data.groupby(key)['Survived'].mean().plot(kind='bar')
    plt.title(key + ' VS Servival Rate')
    plt.ylabel('Servival Rate')
    plt.show()
    
    data.groupby(['Survived',key])['Survived'].count().unstack().plot(kind='bar',stacked='True')
    plt.title(key + ' distribution in Survival ')
    plt.show()
